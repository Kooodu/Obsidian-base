.tours {

      &-side {  
         overflow: hidden;  
  
         position: absolute;  
         height: 52rem;

         width: 100%;  
         border-radius: 3px;

            background: #FFFFFF;  
  
            box-shadow: 0px 4px 4px rgba(0, 0, 0, 0.25), 0px 4px 4px rgba(0, 0, 0, 0.25), 0px 15px 40px rgba(0, 0, 0, 0.15);
