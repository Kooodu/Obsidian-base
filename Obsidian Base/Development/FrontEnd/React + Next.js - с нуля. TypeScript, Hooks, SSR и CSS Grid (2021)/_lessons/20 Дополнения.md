## Обновление SVGR

Обновляем SVGR

`package.json`
```JSON
"devDependencies": {  
	"@svgr/webpack": "^6.1.2",
},
```

Обновляем конфиг некста

`next.config.js`
```JS
module.exports = {  
   images: {  
      domains: ['courses-top.ru']  
   },  
   webpack(config, options) {  
      config.module.rules.push({  
         loader: '@svgr/webpack',  
         issuer: /\.[jt]sx?$/,  
         options: {  
            prettier: false,  
            svgo: true,  
            svgoConfig: {  
               plugins: [{  
                  name: 'preset-default',  
                  params: {  
                     override: {  
                        removeViewBox: false  
                     }  
                  }  
               }],  
            },  
            titleProp: true,  
         },  
         test: /\.svg$/,  
      });  
  
      return config;  
   },  
};
```
